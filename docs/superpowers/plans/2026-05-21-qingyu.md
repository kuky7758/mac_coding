> **面向 AI 代理的工作者：** 必需子技能：使用 superpowers:subagent-driven-development（推荐）或 superpowers:executing-plans 逐任务实现此计划。步骤使用复选框（`- [ ]`）语法来跟踪进度。

**目标：** 实现一个最小可用的匿名短内容发布与浏览应用，单文件纯前端，Local Storage 持久化。

**架构：** 单 HTML 文件内嵌 CSS 与 JavaScript。JS 逻辑按职责分层：数据存取（`loadPosts`/`savePosts`）、UI 渲染（`renderPosts`/`renderEmptyState`）、事件处理（`handlePublish`/`handleInput`）。页面加载时初始化，所有状态变更后重新渲染。

**技术栈：** HTML5、CSS3、原生 JavaScript（无框架）、Local Storage API、Git

```javascript
const STORAGE_KEY = 'qingyu_posts';

function loadPosts() {
  try {
    const raw = localStorage.getItem(STORAGE_KEY);
    return raw ? JSON.parse(raw) : [];
  } catch (e) {
    console.error('读取数据失败', e);
    return [];
  }
}

function savePosts(posts) {
  try {
    localStorage.setItem(STORAGE_KEY, JSON.stringify(posts));
    return true;
  } catch (e) {
    if (e.name === 'QuotaExceededError') {
      alert('存储空间不足，请清理后重试');
    } else {
      alert('保存失败，请重试');
    }
    return false;
  }
}
```

```javascript
const postListEl = document.getElementById('postList');

function formatTime(timestamp) {
  const d = new Date(timestamp);
  const pad = n => String(n).padStart(2, '0');
  return `${d.getFullYear()}-${pad(d.getMonth() + 1)}-${pad(d.getDate())} ${pad(d.getHours())}:${pad(d.getMinutes())}`;
}

function renderPosts(posts) {
  postListEl.innerHTML = '';

  if (posts.length === 0) {
    postListEl.innerHTML = '<div class="empty-state">还没有内容，写下一句话吧</div>';
    return;
  }

  const sorted = [...posts].sort((a, b) => b.timestamp - a.timestamp);

  for (const post of sorted) {
    const card = document.createElement('div');
    card.className = 'post-card';
    card.innerHTML = `
      <div class="post-content">${escapeHtml(post.content)}</div>
      <div class="post-time">${formatTime(post.timestamp)}</div>
    `;
    postListEl.appendChild(card);
  }
}

function escapeHtml(text) {
  const div = document.createElement('div');
  div.textContent = text;
  return div.innerHTML;
}
```

```javascript
const contentInput = document.getElementById('contentInput');
const publishBtn = document.getElementById('publishBtn');
const charCountEl = document.getElementById('charCount');

function updateCharCount() {
  const len = contentInput.value.length;
  charCountEl.textContent = `${len}/280`;
  charCountEl.classList.toggle('over-limit', len > 280);
}

function handlePublish() {
  const content = contentInput.value.trim();
  if (!content) {
    publishBtn.style.transform = 'scale(0.95)';
    setTimeout(() => publishBtn.style.transform = '', 150);
    return;
  }

  const posts = loadPosts();
  const now = Date.now();
  posts.push({ id: now, content, timestamp: now });

  if (savePosts(posts)) {
    renderPosts(posts);
    contentInput.value = '';
    updateCharCount();
  }
}

contentInput.addEventListener('input', updateCharCount);
publishBtn.addEventListener('click', handlePublish);

// 页面初始化
renderPosts(loadPosts());
```
